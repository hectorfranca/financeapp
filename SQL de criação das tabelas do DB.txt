SQL tabela Categoria:
CREATE TABLE `categoria` (
  `Id` int(11) NOT NULL,
  `Nome` varchar(40) COLLATE latin1_bin NOT NULL,
  `Tipo` char(1) COLLATE latin1_bin NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=latin1 COLLATE=latin1_bin;

SQL tabela Registro:
CREATE TABLE `registro` (
  `Id` int(11) NOT NULL,
  `Categoria_id` int(11) NOT NULL,
  `Nome` varchar(60) COLLATE latin1_bin NOT NULL,
  `Data` date NOT NULL,
  `Valor` double NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=latin1 COLLATE=latin1_bin;

SQL para criar FK
ALTER TABLE registro ADD CONSTRAINT fk_categoriaid FOREIGN KEY (Categoria_id) REFERENCES categoria (Id);
